ШАГ 1: Установите зависимости на Linux системе
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

sudo apt-get install -y git unzip openjdk-8-jdk python3-pip autoconf libtool pkg-config zlib1g-dev libncurses5-dev libncursesw5-dev libtinfo5 cmake libffi-dev libssl-dev

pip3 install --user buildozer kivy cython pyperclip
export PATH="$HOME/.local/bin:$PATH"

ШАГ 2: Запустите компиляцию
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

chmod +x build_android.sh
./build_android.sh

ИЛИ вручную:
buildozer android debug

Скопируйте на Android устройство и установите!
